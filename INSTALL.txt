These instructions assume you're using apache2 webserver on ubuntu.


#) install and configure apache
sudo apt-install apache2
sudo ln -s /etc/apache2/mods-available/cgi.load /etc/apache2/mods-enabled/
sudo systemctl restart apache2

#) install perl CGI dependency
sudo apt install libcgi-pm-perl

#) download from github
cd $HOME
git clone https://github.com/penfold42/SDGRemote

#) copy files to webserver
cd SDGRemote
sudo mkdir /var/www/html/sdg
sudo cp -a html/* /var/www/html/sdg/
sudo cp cgi-bin/sdgproxy.pl /usr/lib/cgi-bin/

#) configure your siglent device name/ip
edit /var/www/html/sdg/sdg-devices.json

#) cross your fingers and enjoy:
browse to http://<webserver>/sdg/
